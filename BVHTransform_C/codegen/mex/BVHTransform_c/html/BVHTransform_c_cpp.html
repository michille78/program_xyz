  734   /*
  735    * function euler = CToEuler_c( C,rotateOrder,rotateDirection )
  736    */
  737   static void CToEuler_c(const emlrtStack *sp, const real_T C[9], real_T
  738     rotateOrder, const real_T rotateDirection[3], real_T euler[3])
  739   {
  740     real_T C_K[9];
  741     real_T dv21[3];
  742     int32_T i11;
  743     emlrtStack st;
  744     st.prev = sp;
  745     st.tls = sp->tls;
  746   
  747     /* % xyz  2015.4.5 */
  748     /*  direction cosine matrix  to Euler angle  for multi C */
  749     /*  C: [N*3*3] */
  750     /*        euler = [eu_x,eu_y,eu_z]  ( Nframes*3 ) */
  751     /*  store order of euler angle: the rotate order */
  752     /*  positive:anticlockwise rotate  */
  753     /*  the first and third euler angles scope: [-pi,pi] */
  754     /*  the second euler angles scope: [-pi/2,pi/2] */
  755     /*  unit:rad  */
  756     /*  rotateDirection: anticlockwise(1) or clockwise(-1)  111 */
  757     /* 'BVHTransform_c:842' coder.inline('never'); */
  758     /* 'BVHTransform_c:844' if size(C,3) == 1 */
  759     /* 'BVHTransform_c:846' else */
  760     /* 'BVHTransform_c:847' Nframes = size( C,1 ); */
  761     /* 'BVHTransform_c:848' euler = zeros( Nframes,3 ); */
  762     /* 'BVHTransform_c:849' for k=1:Nframes */
  763     /* 'BVHTransform_c:850' C_K = permute( C(k,:,:),[2,3,1] ); */
  764     memcpy(&C_K[0], &C[0], 9U * sizeof(real_T));
  765   
  766     /* 'BVHTransform_c:851' euler(k,:) = OneCToEuler_c( C_K,rotateOrder,rotateDirection ); */
  767     st.site = &sb_emlrtRSI;
  768     OneCToEuler_c(&st, C_K, rotateOrder, rotateDirection, dv21);
  769     for (i11 = 0; i11 < 3; i11++) {
  770       euler[i11] = dv21[i11];
  771     }
  772   }
