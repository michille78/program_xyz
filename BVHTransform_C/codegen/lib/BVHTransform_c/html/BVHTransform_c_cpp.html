 1456       if (u0 > 0.0) {
 1457         y = RT_PI / 2.0;

 1459         y = -(double)(RT_PI / 2.0);
 1460       } else {
 1461         y = 0.0;
 1462       }
 1463     } else {
 1464       y = atan2(u0, u1);
 1465     }
